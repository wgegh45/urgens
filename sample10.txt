2-gram, 3-gramを作成
from nltk import ngrams
proper_nouns = ["Microsoft", "Tokyo", "Python"]

# 2-gram と 3-gram を生成して文字列化
proper_noun_2grams = ["".join(gram) for gram in ngrams(proper_nouns, 2)]
proper_noun_3grams = ["".join(gram) for gram in ngrams(proper_nouns, 3)]

# 元のリストに追加
proper_nouns.extend(proper_noun_2grams)
proper_nouns.extend(proper_noun_3grams)

print(proper_nouns)
